  For n=20, We would need

  #+BEGIN_SRC python
  def memoize(f):
      d = {}
      def f_(n):
          if n in d: return d[n]
          d[n] = f(n)
          return d[n]
          
      return f_
  
  @memoize
  def T(n):
      if n == 1: return 1
      return sum([T(i) for i in range(1, n)]) + 1

  return T(20)
